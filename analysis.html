def readData(file, label, l = 0, r = -1):
    fin = open(file, "r")

    data = dict()
    for i in label:
        data[i] = []

    for i in fin.readlines()[l:r]:
        arr = [float(x) for x in i.split()]
        if len(label) != len(arr):
            raise BaseException
        for i in range(len(label)):
            data[label[i]].append(arr[i])

    for i in label:
        data[i] = np.array(data[i])
    return data
